# API Reference

:::tip TL;DR
Import `themes` for token values, use `getThemeClassName()` or `applyTheme()` for theme switching.
:::

This page documents all TypeScript exports from `@designgreat/lib-web-ui-design-token`.

---

## Quick Reference

```typescript
import {
  // Most commonly used
  themes,              // Access token values: themes.light, themes.dark
  getThemeClassName,   // Get CSS class for a theme
  applyTheme,          // Apply theme to an element
  
  // Types
  type ThemeName,      // 'light' | 'dark'
} from '@designgreat/lib-web-ui-design-token'
```

---

## Token Access

### `themes` Object

The main way to access token values programmatically:

```typescript
import { themes } from '@designgreat/lib-web-ui-design-token'

// Access light theme tokens
const lightBg = themes.light.dg.color.background.DEFAULT    // "#ffffff"
const lightText = themes.light.dg.color.text.DEFAULT        // "#1d1d1f"

// Access dark theme tokens
const darkBg = themes.dark.dg.color.background.DEFAULT      // "#0c0c0d"
const darkText = themes.dark.dg.color.text.DEFAULT          // "#e8e8e9"

// Spacing
const spacing = themes.light.dg.spacing.px16                // "1rem"

// Typography
const fontFamily = themes.light.dg['font-family'].primary   // "Roboto"
```

### Named Exports: `light` & `dark`

Direct access to specific themes:

```typescript
import { light, dark } from '@designgreat/lib-web-ui-design-token'

const primaryLight = light.dg.color.primary.DEFAULT  // "#0055cc"
const primaryDark = dark.dg.color.primary.DEFAULT    // "#599eff"
```

### `getThemeTokens(theme)`

Get tokens for a specific theme by name:

```typescript
import { getThemeTokens } from '@designgreat/lib-web-ui-design-token'

const tokens = getThemeTokens('light')
const bgColor = tokens.dg.color.background.DEFAULT
```

### `listThemeNames()`

Get array of available theme names:

```typescript
import { listThemeNames } from '@designgreat/lib-web-ui-design-token'

const themes = listThemeNames()  // ['dark', 'light']
```

### `createTheme(base, overrides)`

Create a custom theme with token overrides:

```typescript
import { createTheme } from '@designgreat/lib-web-ui-design-token'

const customTheme = createTheme('light', {
  dg: {
    color: {
      primary: {
        DEFAULT: '#FF0000'  // Override primary color
      }
    }
  }
})
```

## Theme Utilities

### `getThemeClassName(theme)`

Returns the CSS class name for a theme:

```typescript
import { getThemeClassName } from '@designgreat/lib-web-ui-design-token'

getThemeClassName('light')  // '' (light uses :root, no class needed)
getThemeClassName('dark')   // 'dg-theme-dark'
```

**Usage in React:**

```tsx
function App() {
  const [isDark, setIsDark] = useState(false)
  
  return (
    <div className={getThemeClassName(isDark ? 'dark' : 'light')}>
      {/* Your app */}
    </div>
  )
}
```

### `applyTheme(element, theme)`

Applies the theme class to an element:

```typescript
import { applyTheme } from '@designgreat/lib-web-ui-design-token'

// Apply dark theme to document
applyTheme(document.documentElement, 'dark')

// Switch back to light
applyTheme(document.documentElement, 'light')
```

### `isThemeApplied(element, theme)`

Checks if a theme is currently applied:

```typescript
import { isThemeApplied } from '@designgreat/lib-web-ui-design-token'

if (isThemeApplied(document.documentElement, 'dark')) {
  console.log('Dark theme is active')
}
```

### `getThemeSelector(theme)`

Returns the CSS selector for a theme:

```typescript
import { getThemeSelector } from '@designgreat/lib-web-ui-design-token'

getThemeSelector('light')  // ':root'
getThemeSelector('dark')   // '.dg-theme-dark'
```

### `getOppositeTheme(theme)`

Returns the opposite theme name:

```typescript
import { getOppositeTheme } from '@designgreat/lib-web-ui-design-token'

getOppositeTheme('light')  // 'dark'
getOppositeTheme('dark')   // 'light'
```


## TypeScript Types

### `ThemeName`

Union type of available theme names:

```typescript
import type { ThemeName } from '@designgreat/lib-web-ui-design-token'

// ThemeName = 'light' | 'dark'

function setTheme(name: ThemeName) {
  applyTheme(document.documentElement, name)
}
```

### `ThemeToken`

The full token structure for a theme:

```typescript
import type { ThemeToken } from '@designgreat/lib-web-ui-design-token'

function processTheme(tokens: ThemeToken) {
  // TypeScript knows all available properties
  console.log(tokens.dg.color.primary.DEFAULT)
}
```

### `ThemeOverrides`

Partial structure for `createTheme()` overrides:

```typescript
import type { ThemeOverrides } from '@designgreat/lib-web-ui-design-token'

const overrides: ThemeOverrides = {
  dg: {
    color: {
      primary: {
        DEFAULT: '#custom-color'
      }
    }
  }
}
```

## Usage Examples

### Theme Toggle Component

```tsx
import { useState } from 'react'
import { 
  getThemeClassName, 
  type ThemeName 
} from '@designgreat/lib-web-ui-design-token'

function ThemeToggle() {
  const [theme, setTheme] = useState<ThemeName>('light')
  
  return (
    <div className={getThemeClassName(theme)}>
      <button onClick={() => setTheme(t => t === 'light' ? 'dark' : 'light')}>
        Current: {theme}
      </button>
    </div>
  )
}
```

### Dynamic Styling

```tsx
import { themes } from '@designgreat/lib-web-ui-design-token'

function DynamicBox({ isDark }: { isDark: boolean }) {
  const t = isDark ? themes.dark : themes.light
  
  return (
    <div style={{
      backgroundColor: t.dg.color.background.DEFAULT,
      color: t.dg.color.text.DEFAULT,
      padding: t.dg.spacing.px16
    }}>
      Dynamic content
    </div>
  )
}
```

### Theme-Aware Utility

```typescript
import { themes, type ThemeName } from '@designgreat/lib-web-ui-design-token'

function getContrastColor(theme: ThemeName, isError: boolean) {
  const t = themes[theme]
  return isError 
    ? t.dg.color.semantic.error.DEFAULT 
    : t.dg.color.text.DEFAULT
}
```
